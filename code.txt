        def __init__(self, brand, model, year):                 
          self.brand = brand
          self.model = model

        def accelerate(self, acceleration):
          self.speed += acceleration    
        def brake(self, deceleration):
          if self.speed - deceleration >= 0:        
            self.speed -= deceleration
          else:
            self.speed = 0

          self.direction = direction
